print "Testing if-else statements..."

x = 10
y = 5

if (x > y) {
    print "x is greater than y"
}

if (y > x) {
    print "This line should NOT appear (y > x is false)"
} else {
    print "y is NOT greater than x (else block executed)"
}

is_active = true
if (is_active == true) {
    print "System is active."
} else {
    print "This else should NOT appear."
}

if (not is_active) {
    print "This should NOT appear (not is_active is false)"
} else {
    print "System is confirmed active (else block for 'not is_active')."
}
